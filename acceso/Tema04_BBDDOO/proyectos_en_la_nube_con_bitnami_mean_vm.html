Proyectos en la nube con Bitnami MEAN VM

Descargamos una máquina virtual de bitnami

Documentación: https://docs.bitnami.com/virtual-machine/infrastructure/mean/

La máquina: https://bitnami.com/stack/mean

Configurando el teclado en español

sudo apt-get install console-data
sudo apt-get install console-setup
sudo apt-get install keyboard-configuration

Login en mongo

$ mongo admin -u root -p bitnami
> show databases # ver las bases de dato
> use local
> show collections
> use test # usa la base de datos, si no existe la crea
>

Configurando la máquina virtual

Configuramos SSHD para que permita identificación con Password y puerto 2222
En el fichero /etc/ssh/sshd_config:

  • Port 2222
  • ChallengeResponseAuthentication yes
  • PasswordAuthentication yes
$ sudo su
# mv /etc/init/ssh.conf.back /etc/init/ssh.conf
# mkdir /var/run/sshd
# chmod 0755 /var/run/sshd
# ufw allow 2222
# start ssh

Conectando a la interfaz Web RockMongo para gestionar nuestra base de datos Mongo

Hay que hacer un túnel SSH para enrutar la conexión Web y hacerla local.

$ ssh -N -L 8888:127.0.0.1:80  bitnami@192.168.1.49 -p 2222

Creando una rutina para la carga inicial de datos

Esta rutina nos sirve para aprender a usar el driver MongoClient y probar la URI de conexión:

var MongoClient = require('mongodb').MongoClient;
var cat = require('cat');

var urlConnection = 'mongodb://root:bitnami@localhost:27017/mydb?authSource=admin';
var myCollection = "alumnos";

var datosJSON = [
    {
        "apellido": "PEREZ",
        "id": 1,
        "nombre": "PEPE"
    },
    {
        "apellido": "MARTINEZ",
        "id": 2,
        "nombre": "DANIEL"
    }
];

MongoClient.connect(urlConnection, {uri_decode_auth:1}, function(err0, db) {
  if (!err0) {
    console.log("Connectado al servidor");    
    var dbCollection = db.collection(myCollection);
    dbCollection.insert(datosJSON,function(err2,result){
        if(!err2) {
            console.log("Éxito: "+result.ops.length+" alumnos insertados!");
        } else {
            console.log("No pude insertar los datos.");
        }
    });
    db.close();
  } else {
      console.log('No puedo conectar al servicio MongoDB'+err0);
  }  
});
